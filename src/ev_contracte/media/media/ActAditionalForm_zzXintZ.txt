class ContractDetailView(DetailView):
    model = Contract
    template_name = 'contr_clienti/contract_detail.html'
    context_object_name = 'obj_contract'
    # queryset = get_context_data

    def get_context_data(self, **kwargs):
        pk = self.kwargs['pk']
        pk1 = self.kwargs['pk1']
        context = super(ContractDetailView, self).get_context_data(**kwargs)
        context['obj_actaditional'] = ActAditional.objects.all(id = self.kwargs['pk1'])
        context['obj_contract'] = self.object
        return context

    def get_success_url(self):
        return reverse('contract-detail', kwargs={'pk': self.object.pk, 'pk1': self.object.pk1})
